**Bubblesort 函數呼叫 副函式執行**

### 若有指定數字及陣列大小
```c
#include<stdio.h>

#define MAX 6

void bubble(int * , int);

int main(){

    int data[MAX] = {11,12,10,15,1,2};
    bubble(data , MAX);
    printf("Bubble Sort: ");
    for(int i=0 ; i < MAX ; i++){
        printf("[%d]" , data[i]);
    }
    printf("\n");
    return 0;
}

void bubble(int data[] , int count){

    int i, j, temp;

    for(j=count;j>1;j--){
        for(i=0;i<j-1;i++){
            if(data[i+1] < data[i]){
                temp = data[i];
                data[i] = data[i+1];
                data[i+1] = temp;
            }
        }
    }
}
```
