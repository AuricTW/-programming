//找出陣列的最大值
void maxElement(int *eles, int *index){

    int maxValue = eles[0];
    *index = 0;

    for(int i=1; i<LEN; i++){
        if(eles[i] > maxValue){
            maxValue = eles[i];
            *index = i;//最大值的陣列索引
        }
    }

    int temp = eles[0];
    eles[0] = maxValue;

}
